## Basic Example

PeriodSelector provides a simple way for users to switch between different time periods in charts.

```jsx
function BasicPeriodSelector() {
  const [selectedPeriod, setSelectedPeriod] = useState('1D');
  
  const periods = [
    { label: '1D', value: '1D' },
    { label: '1W', value: '1W' },
    { label: '1M', value: '1M' },
    { label: '3M', value: '3M' },
    { label: '1Y', value: '1Y' },
    { label: 'ALL', value: 'ALL' },
  ];
  
  return (
    <PeriodSelector
      periods={periods}
      selectedPeriod={selectedPeriod}
      onPeriodChange={setSelectedPeriod}
    />
  );
}
```

## With Chart Integration

Integrate PeriodSelector with a chart to dynamically update the displayed data.

```jsx
function ChartWithPeriodSelector() {
  const [selectedPeriod, setSelectedPeriod] = useState('1M');
  
  const dataByPeriod = {
    '1D': [100, 105, 103, 108, 106, 110],
    '1W': [100, 102, 105, 103, 108, 106, 110, 108, 112],
    '1M': [100, 98, 102, 105, 103, 108, 106, 110, 108, 112, 115, 113, 118],
    '3M': [90, 95, 92, 98, 100, 102, 105, 103, 108, 106, 110, 108, 112, 115, 113, 118, 120, 122],
    '1Y': [80, 85, 88, 90, 92, 95, 98, 100, 105, 110, 108, 112, 115, 118, 120, 125, 130],
    'ALL': [50, 60, 70, 80, 85, 88, 90, 92, 95, 98, 100, 105, 110, 115, 120, 125, 130, 135],
  };
  
  const periods = Object.keys(dataByPeriod).map(key => ({ label: key, value: key }));
  
  return (
    <VStack gap={3}>
      <PeriodSelector
        periods={periods}
        selectedPeriod={selectedPeriod}
        onPeriodChange={setSelectedPeriod}
      />
      <LineChart
        height={250}
        series={[
          {
            id: 'price',
            data: dataByPeriod[selectedPeriod],
          },
        ]}
        curve="monotone"
        showYAxis
        yAxis={{
          showGrid: true,
        }}
      />
    </VStack>
  );
}
```

Customize the appearance of the PeriodSelector.

```jsx

  periods={[
    { label: '1H', value: '1H' },
    { label: '24H', value: '24H' },
    { label: '7D', value: '7D' },

  selectedPeriod="24H"
  onPeriodChange={(period) => console.log('Selected:', period)}

  inactiveColor="rgb(156, 163, 175)"

/>
```

## With Loading State

Show loading state while fetching new data for a period.

```jsx
function LoadingPeriodSelector() {
  const [selectedPeriod, setSelectedPeriod] = useState('1D');
  const [isLoading, setIsLoading] = useState(false);
  
  const handlePeriodChange = async (period) => {
    setIsLoading(true);
    setSelectedPeriod(period);
    
    // Simulate data fetching
    await new Promise(resolve => setTimeout(resolve, 1000));
    setIsLoading(false);
  };
  
  return (
    <VStack gap={3}>
      <PeriodSelector
        periods={[
          { label: '1D', value: '1D' },
          { label: '1W', value: '1W' },
          { label: '1M', value: '1M' },
        ]}
        selectedPeriod={selectedPeriod}
        onPeriodChange={handlePeriodChange}
        disabled={isLoading}
      />
      {isLoading && <Text>Loading data...</Text>}
    </VStack>
  );
}
```
