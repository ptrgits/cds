## Basic Example

Line charts display data points connected by straight line segments, ideal for showing trends over time.

```jsx
<LineChart
  height={250}
  series={[
    {
      id: 'prices',
      data: [10, 25, 20, 35, 30, 45, 40, 50, 48, 55],
    },
  ]}
/>
```

## Multiple Lines

Display multiple data series for comparison.


## Interactive with Scrubber

Enable scrubbing for detailed value inspection.

```jsx
<LineChart
  enableScrubbing
  height={250}
  series={[
    {
      id: 'portfolio',
      data: [1000, 1200, 1150, 1400, 1380, 1500, 1450, 1600, 1750, 1700, 1850, 2000],
    },
  ]}
  curve="monotone"
  showYAxis
  showArea
  yAxis={{
    showGrid: true,
    formatValue: (value) => `$${value.toLocaleString()}`,
  }}
  renderPoints={() => true}
>
  <Scrubber />
</LineChart>
```

## Conditional Points

Show points conditionally based on data characteristics.

```jsx
function ConditionalPointsChart() {
  const data = [20, 35, 15, 45, 40, 55, 30, 65, 60, 70];
  const maxValue = Math.max(...data);
  const minValue = Math.min(...data);
  
  return (
    <LineChart
      height={250}
      series={[
        {
          id: 'conditional',
          data: data,
        },
      ]}
      curve="monotone"
      renderPoints={(datum) => {
        // Show points only for min and max values
        return datum.value === maxValue || datum.value === minValue;
      }}
      showYAxis
      yAxis={{
        showGrid: true,
      }}
    />
  );
}
```

## Time Series

Display time series data with proper formatting.

```jsx
function TimeSeriesLineChart() {
  const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug'];
  const data = [45, 52, 48, 61, 58, 65, 70, 68];
  
  return (
    <LineChart
      height={250}
      series={[
        {
          id: 'monthly-sales',
          data: data,
        },
      ]}
      curve="monotone"
      showYAxis
      showXAxis
      yAxis={{
        showGrid: true,
        formatValue: (v) => `${v}k`,
      }}
      xAxis={{
        data: months,
      }}
      showArea
    />
  );
}
```
